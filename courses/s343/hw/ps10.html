10 11/18 ∗3 Implement your own version of CIsim, specific to data drawn from a normal distribution. The version below is complete if we are interested only a 95% confidence level. Modify it so as to make adaptable to any specified level.
 myCIsim <- function(n=20, samples=10000, mu=0, sd=1) {
  insideInt <- replicate(samples, {
   x <- rnorm(n);
   this.t.test <- t.test(~x);
   this.t.test$conf.int[1] < mu & mu < this.t.test$conf.int[2]
  })
  return ( sum(insideInt) / samples )
 }
